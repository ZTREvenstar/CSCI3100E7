CREATE TABLE customer (id INT NOT NULL, password VARCHAR(40) NOT NULL), PRIMARY KEY (id))

CREATE comment( id INT, dishID INT, customerID INT, content VARCHAR(100) NOT NULL, likeNum INT, PRIMARY KEY(id))

////////////////////////////////////////////////
CREATE TABLE orderinfo
             (orderID INT, 
              customerID INT,
              dishID INT,
              time DATETIME,
              status INT,
              PRIMARY KEY (orderID) 
             ) 
 // FOREIGN KEY (customer_id) REFERENCES customer,



/////////////////////////////////////////////////////////////

CREATE TABLE personalinfo
             (customer_id INT NOT NULL, 
              email_address VARCHAR(64),
              profilepicture_link VARCHAR(64),
              profile_name VARCHAR(20),
              PRIMARY KEY (customer_id),
              FOREIGN KEY (customer_id) REFERENCES customer
             );

CREATE TABLE favourites
             (customer_id INT NOT NULL, 
              dish_id INT NOT NULL,
              PRIMARY KEY (customer_id, dish_id),
              FOREIGN KEY (customer_id) REFERENCES customer,
              FOREIGN KEY (dish_id) REFERENCES dish
             );

customer          id (primary key)  |  username |  password

comment         id (primary key)  |  dishID  |  customerID  |  content  | likeNum
